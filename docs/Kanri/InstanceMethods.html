Determines if an action can be performed on the given target by the user

If the instance calling this method responds to the #user method, and no user is explicitly passed, the output of the #user method will be used instead.

def can?(action, target, user: nil)
    user ||= respond_to?(:user) ? self.user : nil
    Kanri.roles[self.class]
         .select { |role| role.include? user, target }
         .any? { |role| role.can? user, action, target }
end
